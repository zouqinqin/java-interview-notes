## 移位运算符
- 1. << 左移
     3 << 2
     // 3 的二进制 = 0000 0011
     // 左移2位    = 0000 1100 = 12
     // 规律：左移n位 = 乘以2的n次方
     // 3 × 2² = 3 × 4 = 12 ✅
- 2. >> 右移（有符号）
     12 >> 2
     // 12 的二进制 = 0000 1100
     // 右移2位     = 0000 0011 = 3
     // 规律：右移n位 = 除以2的n次方
     // 12 ÷ 2² = 12 ÷ 4 = 3 ✅
  - 3. >>> 无符号右移
       -1 >>> 1
       // -1 的二进制 = 1111 1111 1111 1111（32个1）
       // 无符号右移1位 = 0111 1111 1111 1111
       // 结果 = 2147483647（Integer最大值）

         // 和 >> 的区别：
         // -1 >> 1   // = -1  左边补1，符号不变
         // -1 >>> 1  // = 2147483647  左边强制补0
